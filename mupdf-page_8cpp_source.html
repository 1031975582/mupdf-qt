151 QImage Page::renderImage() const
152 {
153  fz_pixmap *pixmap = NULL;
154  unsigned char *samples = NULL;
155  int width = 0;
156  int height = 0;
157  int size = 0;
158 
159  // apply scale and rotate
160  fz_matrix transform;
161  fz_rotate(&transform, d->rotation);
162  fz_pre_scale(&transform, d->scaleX, d->scaleY);
163 
164  // get transformed page size
165  fz_rect bounds;
166  fz_bound_page(d->document, d->page, &bounds);
167  fz_transform_rect(&bounds, &transform);
168  fz_irect bbox;
169  fz_round_rect(&bbox, &bounds);
170 
171  // render to pixmap
172  fz_device *dev = NULL;
173  fz_try(d->context)
174  {
175  pixmap = fz_new_pixmap_with_bbox(d->context, fz_device_bgr(d->context), &bbox);
176  samples = fz_pixmap_samples(d->context, pixmap);
177  width = fz_pixmap_width(d->context, pixmap);
178  height = fz_pixmap_height(d->context, pixmap);
179  size = width * height * 4;
180  if (!d->transparent) {
181  if (d->b >= 0 && d->g >= 0 && d->r >= 0 && d->a >= 0) {
182  // with user defined background color
183  clear_samples_with_value(samples, size, d->b, d->g, d->r, d->a);
184  } else {
185  // with white background
186  fz_clear_pixmap_with_value(d->context, pixmap, 0xff);
187  }
188  }
189  dev = fz_new_draw_device(d->context, pixmap);
190  fz_run_display_list(d->display_list, dev, &transform, &bounds, NULL);
191  }
192  fz_always(d->context)
193  {
194  if (dev) {
195  fz_free_device(dev);
196  }
197  dev = NULL;
198  }
199  fz_catch(d->context)
200  {
201  if (pixmap) {
202  fz_drop_pixmap(d->context, pixmap);
203  }
204  pixmap = NULL;
205  }
206 
207  // render to QImage
208  QImage image;
209  if (NULL == pixmap) {
210  return image;
211  }
212  // No copy before return
213  info_s *info = new info_s;
214  info->context = d->context;
215  info->pixmap = pixmap;
216  image = QImage(samples, // no deep copy here
217  width, height, QImage::Format_ARGB32, imageCleanupHandler, info);
218  return image;
219 }
